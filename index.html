async function sendMessageToGemini(userMessage) { console.log("🚀 جاري الاتصال بجوجل Gemini الحقيقي..."); // 1. الرابط الرسمي والصحيح 100% من جوجل (تعديل الرابط هنا) const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${GEMINI_API_KEY}`; try { const response = await fetch(url, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ contents: [{ parts: [{ text: userMessage }] }] }) }); const data = await response.json(); // 2. استخراج الجواب الحقيقي القادم من سيرفرات جوجل let aiReply = data.candidates[0].content.parts[0].text; console.log("✅ تم استلام الجواب بنجاح!"); return aiReply; // هذا الجواب اللي راح يعرضه موقعك للمستخدم } catch (error) { console.error("❌ حدث خطأ أثناء الاتصال بالـ AI:", error); return "تأكد من كتابة مفتاح الـ API بشكل صحيح في ملف config.js"; } }
